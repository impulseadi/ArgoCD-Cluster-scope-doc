Kubernetes Yaml files:

1.Pod Yaml file:

apiVersion: v1
kind: Pod
metadata:
  name: adi-pod
  namespace: adi
  labels:
    app: adi-nginx
spec:
  containers:
-name: adi-nginx
image: nginx: latest
ports:
  - containerPort: 80

  namespace: adi

    app: adi-ipl

  type: ClusterIP
